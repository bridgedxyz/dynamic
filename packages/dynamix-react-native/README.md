## Installation

```sh
npm install @dynamix/react-native
```

## Usage

```tsx
import dynamix, { DynamixProvider, Webview } from "@dynamix/react-native";

function App() {
  return (
    <DynamixProvider>
      <Webview />
    </DynamixProvider>
  );
}
```
